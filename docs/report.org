def moving_average(x, w):
    return np.convolve(x, np.ones(w), 'same') / w

rewards = np.load('../src/log/10000_rewards.npy')
episodes = np.arange(0, len(rewards))
fig, ax = plt.subplots(1,1, facecolor='w')
ax.plot(episodes, rewards, label='Reward')
ax.plot(episodes, moving_average(rewards, 100), label='Rolling average')
